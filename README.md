namespace MyCalcLibrary
{
    public class MyCalc
    {
        public int Sum(int x, int y)
        {
            return x + y;
        }
        public int Minus(int x, int y)
        {
            return x - y;
        }
        public int Mult(int x, int y)
        {
            return x * y;
        }                                                             
    }
}
